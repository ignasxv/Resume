\newcommand{\tableEnv}[2]{%
 \begin{rSection}{\small \normalfont\bfseries #1} % Use \normalfont and \bfseries to remove italics

      \begin{tabularx}{\linewidth}{@{} >{\bfseries}l @{\hspace{6ex}} X @{}}
        #2 % Print the content inside the tabular environment
      \end{tabularx}
    \end{adjustwidth}
  \end{rSection}
}
